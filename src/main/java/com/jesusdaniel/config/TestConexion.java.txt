package com.jesusdaniel.config;

import java.sql.Connection;
import java.sql.ResultSet;
import java.sql.Statement;

public class TestConexion {
    public static void main(String[] args) {
        try (Connection conn = ConexionMySQL.getConnection()) {
            if (conn != null) {
                System.out.println("‚úÖ Conexi√≥n exitosa a la base de datos.");
                String sql = "SELECT * FROM clientes";

                try (Statement stmt = conn.createStatement();
                     ResultSet rs = stmt.executeQuery(sql)) {

                    System.out.println("üìã Registros de la tabla clientes:");
                    while (rs.next()) {
                        int id = rs.getInt("id");
                        String nombre = rs.getString("nombre");
                        String correo = rs.getString("correo");
                        String telefono = rs.getString("telefono");
                        String direccion = rs.getString("direccion");

                        System.out.println(id + " | " + nombre + " | " + correo + " | " + telefono + " | " + direccion);
                    }
                }
            } else {
                System.out.println("‚ùå No se pudo establecer la conexi√≥n.");
            }
        } catch (Exception e) {
            System.out.println("‚ùå Error al conectar: " + e.getMessage());
        }
    }
}
